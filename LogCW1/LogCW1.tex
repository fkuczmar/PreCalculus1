\begin{question}    \label{Ex0:LogFfdf}
\begin{enumerate}
\item Explain the \emph{meaning} of $\log_2(100)$. Approximate its value \emph{without} using a calculator. Explain your reasoning.

\item Explain the \emph{meaning} of $\log_2(1/100)$. Use the result of part (a) to approximate its value \emph{without} using a calculator. Explain your reasoning.

\item Explain the \emph{meaning} of $\log_{100}(2)$. Use the result of part (a) to approximate its value \emph{without} using a calculator. Explain your reasoning.

\item Explain the \emph{meaning} of $\log_{100}(1/2)$. Use th result of part (a) to approximate its value \emph{without} using a calculator. Explain your reasoning.

\item Use the result of part (a) to approximate $\log_2(10)$. Explain your reasoning.

\item What is the {\bf meaning} of $2^{\log_2 100}$ ? Determine its value \emph{without} a calculator. Explain your reasoning.

\item What is the {\bf meaning} of $\log_2 (2^{100})$ ? Determine its value \emph{without} a calculator. Explain your reasoning.

\end{enumerate}

\end{question}
