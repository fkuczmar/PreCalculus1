      f(x) = 8x^2 -14x + 3.
\]

\begin{enumerate}
\item Evaluate $f(-2)$.

\item Solve the equation $f(x)=3$.

\item Solve the equation 
\[ 
   f(x) = f(4).
\]

\item Find all inputs to the function $f$ that are $26$ less than their outputs. Check your answer(s).

Do not use a caculator.

\end{question}

\begin{question}  \label{Qf34k4df3435r4}
Let
\[
        w(x) = 5-11x .
\]

\begin{enumerate}
\item  Solve the equation
\[
    w(x) - w(x+3) = 24 .
\]
Write the solution as a set.

\item Solve the equation
\[
    w(x) - w(x+3) = -33.
\]
Write the solution as a set.

\item Doubling the input to $w$ decreases the output by $14$. Find all possibilities for the input.

\item Solve the inequality $w(x)< w(45)$. Write the solution as a set.

\item Solve the inequality $w(x) < w(c)$ for $x$ in terms of $c$. Write the solution as a set.

\end{enumerate}
\end{question}


\begin{question}  \label{Qe34rsr344r}
Let
\[
     f(x) = x^2 + 6x + 1 .
\]

Answer the following questions. When solving a quadratic equation either complete the square or factor. Do \emph{not} use the quadratic formula. Write solutions to equations as solution sets using set-builder notation. No need for explanations. 

(a) Evaluate $f(1)$.

(b) Solve the equation $f(x)=10$.

(c) Solve the equation $f(x)=f(10)$.

(d) Solve the equation $f(x) = 2 f(-1)$.

(e) Simplify the expression $f(w) - f(w-3)$.

(f) Simplify the expression
\[
      \frac{f(w) - f(3)}{w-3} .
\]

(g) Solve the equation
\[
           f(t+3) - f(t) = 10 .
\]

(h) Solve the equation
\[
           \frac{f(x) - f(3)}{x-3} = 10.
\]
\end{question}
