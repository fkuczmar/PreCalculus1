\[
    \{w | w= \answer{6776} \} .

    \frac{4}{11} - \frac{8}{9} = -\frac{52}{99}

\[
       \{x | x= \answer{-542 \pm \sqrt{40}} \}.

       \{s | s= \answer{0, 45/7} \}.

     2^{-3} + 36^{-1/2} = \answer{\frac{7}{24}}
