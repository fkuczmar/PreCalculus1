\section{Discrete Functions}

\begin{question} \label{QMfmdf33}
The graph of the function $y=f(x)$ is just the one point  with coordinates $(-7,3)$. Find the coordinates of all points (if any) on the graphs of each function below. Explain your reasoning.

\begin{enumerate}
\item $h_1(x) = f \left( \frac{x}{4} \right)$

\item $h_2(x) = f(4x)$.

\item $h_3(x) = f(x-12)$.

\item $h_4(x) = f(x^2 - 6x)$

\item $h_5(x) = f(2^x)$

\item $h_6(x) = f(\log_2 x)$

\item $h_7(x) = f(\sqrt{x})$

\item $h_8(x) = f \left( \frac{3-7x}{x-3} \right)$

\item $h_9(x) = f(b + mx) \, , \, m\neq 0$
\end{enumerate}

\end{question}


\begin{question} \label{QmQnwenSW}

The point $(a,b)$ lies on the graph of the function $y=f(x)$. Find the coordinates of points that lie on the graphs of each of the following functions. Explain your reasoning from first principles.

\begin{enumerate}
\item $g_1(x) = f\left( \frac{1}{2}x \right)$

\item $g_2(x) = f(2x)$

\item $g_3(x) = f\left( \frac{23}{13}x \right)$

\item $g_4(x) = f(-x)$

\item $g_5(x) = f\left( -\frac{7}{12}x \right)$

\item $g_6(x) = f(x-12)$

\item $g_7(x) = f(x+5)$

\item $g_8(x) = f(3(x+5)$

\item $g_9(x) = f(-3x + 12)$

\item $g_{10}(x) = f\left(\frac{13}{7} \left( x-5 \right) \right)$

\item $g_{11}(x) = f(mx+c) \, , \, m\neq 0 $
\end{enumerate}
\end{question}

\section{An Application}

\begin{question} \label{QKdfm3re32}
The function
\[
       G = f(m) \, , \, 0\leq m \leq 200 ,
\]
expresses the number of gallons of gas in your car as a function of the trip-odometer reading (in miles) during a 200-mile trip.

Let 
\[
    L = h(s)
\]
be the function that expresses the number of liters of gas in your car as a function of the trip-odometer reading (\emph{in km}) during the trip.

We'll assume there are exactly $1.6$km in one mile and exactly $4$ liters in one gallon. We'll also assume that $f(150) = 4$.

\begin{enumerate}
\item Write the domain of the function $h$ in set-builder notation.

\item Find the coordinates of a point on the graph of the function $G=h(s)$.

\item Express the function $h$ in terms of $f$. Use the correct input and the correct output.

\item Express $f$ in terms of $h$. Use the correct input and the correct output.

\item Use the graph of the function $G=f(m)$ below to sketch (by hand) a graph of the function $L=h(s)$.
