\begin{question} \label{Qpf3b}
The graph of the function $y=f(x)$ is a line through the point $(5,8)$ with slope $3/7$.

\begin{enumerate}
\item Evaluate $f(5)$.

\item Evaluate $f(36)$. Start by drawing a graph.

\item Find an expression for $f(x)$. Do \emph{not} use $y=mx+b$.
\end{enumerate}

\end{question}

\begin{question} \label{QPfef30}
For prices betweeen $\$4$/burger and $\$12$/burger, the equation
\[
     25p + 4q = 420
\]
relates the price $p$ (measured in dollars/burger) to the average number of burgers sold/day at Five Guys in Shoreline.

\begin{enumerate}
\item What happens to the average number of burgers sold/day if Five Guys increases the price by $\$0.50$/burger?

\item How could Five Guys increase the average number of burgers they sell by $12$ burgers/day? 
\end{enumerate}

\end{question}

\begin{question} \label{Qpffef0000}
Between altitudes of $30$ and $55$ km above the surface of the planet Krypton, the pressure (meausred in atmospheres) is a linear function of altitude. Suppose the presssure is $0.8$ atm at an altitude of $30$km and $0.5$ atm at an altitude of $55$ km.

\begin{enumerate}
\item Find a function
\[
        h = f(P) 
\]
that expresses altitude (measured in km) in terms of the the pressure (measured in atm). Start by sketching the appropriate graph.
Include the correct domain. Work with fractions, not decimals.

\item Find the pressure at an altitude of $45$ km.
\end{enumerate}
\end{question}
