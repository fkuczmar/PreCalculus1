\section{Translating Between Math and English}

\begin{question} \label{QMDM3er3r3}

The function 
\[
    g = f(v) = 40-\frac{1}{25}\left(v-50\right)^{2} \, , \, 35\leq v \leq 70 ,
\]
expresses the gas mileage (in miles/gal) of a car in terms of its speed.

\begin{enumerate}

\item Use the graph of the function $f$ below to find approximate answers to the following questions. Answer each question with a complete sentence.

\item Then use algebra to find the exact answers. Again, answer each question with a complete sentence.

\end{enumerate}

\begin{onlineOnly}

\end{center}
\end{onlineOnly}

\href{https://www.desmos.com/calculator/rmpah0rcez}{142: Gas Mileage 45}

\begin{enumerate}
\item At what speed(s) does the car get $35$ miles/gal? Find the exact speed(s) without a calculator. Then use a calculator to approximate the speed(s) to the nearest tenth of a mile/hour.

\item At what speed(s) is the gas mileage the same as it is at a speed of $38$ miles/hour. Do \emph{not} use a calculator.

\item At what speed(s) does increasing the speed by $10$ miles/hour leave the gas mileage unchanged. 

\item  At what speed(s) does decreasing the speed by $10$ miles/hour increase the gas mileage by $8$ miles/gal?
\end{enumerate}
\end{question}
