\begin{question}  \label{QFun25}

The function
\[
   g = f(v) , 35 \leq v \leq 70 ,
\]
expresses the gas mileage (in miles/gal) of a car in terms of its speed (in miles/hour). Translate each of the following statements into everyday Engish or into mathematics.

\begin{enumerate}

\item Solve the equation $f(v) = 40$.

\item Evaluate $f(40)$.

\item Determine the speed(s) at which the car gets $32$ miles/gal.

\item At what speeds does the car get at most $32$ miles/gal?

\item Solve the equation $f(v)=f(40)$.

%\item Solve the inequality $f(v)\leq 38$.

\item Solve the inequality $f(v) geq f(40)$.

\item Solve the equation $f(v)=f(v-8) + 3$.

\end{enumerate}
\end{question}

\section{Solving Equations}

\begin{question} \label{Qkdfk4r3}
Let 
\[
    f(x) = x^2 - 10x + 1.
\]
\begin{enumerate}
\item Simplify the expression
\[
        \frac{f(x) - f(3)}{x-3} .
\]

\item Solve the equation 
\[
        \frac{f(x) - f(3)}{x-3}  = 4.
\]

\item Solve the equation 
\[
     f(x) = 1 - f(-1) .
\]

\item Solve the equation
\[
    f(x) = f(-x) .
\]

\item Solve the equation
\[
    f(x) = - f(x).
\]

\item Solve the equation
\[
     f(x) = 2f(x).
\]



\end{enumerate}



\end{question}
